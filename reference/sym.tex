\chapter*{List of Notations}
\addcontentsline{toc}{chapter}{List of Notations}
\begin{thenotation}
\item[{\makebox[\thenotationmargin][l]{\bf Symbol}}] \hfill {\bf Meaning}
\hfill {\bf Page}
\notationspace
\item[$A \hookrightarrow B$] Monomorphism from $A$ to $B$
\notationspace
\item[$|\cC|$] Set of objects of $\cC$
\item[$\cC^{\rm op}$] Opposite category of $\cC$
\item[$\cC^+$] Same as $\cC$ \leaderfill \pageref{def-free-fixed-category}
\item[$\cC^-$] Same as $\cC^{\rm op}$ \leaderfill
\pageref{def-free-fixed-category}
\item[$\cC^\bot$] One-object-one-morphism category \leaderfill
\pageref{def-free-fixed-category}
\item[$\cC^\top$] Discrete category of $\cC$ \leaderfill
\pageref{def-free-fixed-category}
\item[$\cC\times\cD$] Product category of $\cC$ and
$\cD$
\item[\bf Cat] Category of small categories
\item[\bf CAT] Category of large categories
\item[${\bf CFE}(\Gamma)$] Set of closed functorial expressions \leaderfill
\pageref{def-CFE}
\item[${\bf CMOD}(\pair{\Gamma,\Delta,\Psi})$] Category of CSL models
\leaderfill \pageref{def-CMod}
\item[\bf CSig] Category of CSL signatures \leaderfill \pageref{def-csig}
\notationspace
\item[$F \natrightarrow G$] Natural transformation
\item[$F \adjoint G$] $F$ is a left adoint of $G$
\item[${\bf FE}(\Gamma,\cC)$] Category of functorial expressions \leaderfill
\pageref{def-fe}
\item[${\bf FE}(\Gamma,V)$] Set of functorial expressions over $V$ \leaderfill
\pageref{def-fe-v}
\item[${\bf Funct}(\cC,\cD)$] Functor category
\notationspace
\item[$\Hom{\cC}{A}{B}$] Set of morphisms from $A$ to $B$ of
category \cC
\notationspace
\item[${\bf I}_A$] Identity morphism of $A$
\notationspace
\item[${\bf Nat}(F,G)$] Set of natural transforamtions $F \natrightarrow G$
\notationspace
\item[${\bf pCMod}(\Gamma)$] Category of pre-CSL models \leaderfill
\pageref{def-pCMod}
\item[\bf pCSig] Category of pre-CSL signatures \leaderfill
\pageref{def-pCSig}
\notationspace
\item[$S^\ast$] Set of sequences of elements in $S$
\item[\bf Set] Category of small sets
\notationspace
\item[\bf Var] Set of variances \leaderfill \pageref{def-variance-set}
\item[\bf Vr] Countable set of variables \leaderfill \pageref{def-Vr}
\item[$v\bullet w$] Composition of variances \leaderfill
\pageref{def-var-comp-op}
\item[$v\lleq w$] Ordering on variances \leaderfill \pageref{def-var-ord}
\end{thenotation}
